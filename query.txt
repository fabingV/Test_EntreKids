SELECT DISTINCT P.EP_NOMBRE, SUM(T.ET_TOTAL) AS VENTAS_TOTALES_MES, IF (I.EI_ID = PA.EI_ID, 'PAQUETE','EXPERIENCIA') AS CLAVE
FROM ekt_transaccion T, ekt_item I, ekt_actiividad_evento AE, ekt_actividad A, ekt_proveedor P, ekt_paquete PA
WHERE P.EP_ID = A.EP_ID AND A.EA_ID = AE.EA_ID AND AE.EAE_ID = I.EAE_ID AND I.ET_ID = T.ET_ID AND T.ET_CREATED BETWEEN '2022-08-01' AND '2022-08-31'
GROUP BY P.EP_NOMBRE,CLAVE
ORDER BY P.EP_NOMBRE;
